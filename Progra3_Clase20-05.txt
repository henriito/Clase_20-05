//Crear Clase
class HashTable {
    constructor(size){
        this.data = new Array(size);
    }

//Funcion que genera el ID
    hashMethod(key){
        let hash = 0;
        for (let i=0; i< key.length; i++){
            hash += (key.charCodeAt(i) * i) % this.data.length;
        }
        return hash % this.data.length;
    }

    set(key, value){
        const address = this.hashMethod(key); //genera un id y lo guarda en address
        if (!this.data[address]){
            this.data[address] = []; //Si no existe, crea un arreglo vacÃ­o
        }
        this.data[address].push([key,value]);
        return this.data;

}//Fin de la Clase
